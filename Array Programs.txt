1. WAP to ask user to enter n numbers and print highest/lowest number.
ascending order: low to high
descending order: high to low.

suppose, you have taken array a[10] and user gave 10 random numbers input.After which you arranged them in ascending order.

#include<stdio.h>
#include<conio.h>
void main()
{
	
	int size=30;
	int a[size];
	int i=0,j=0,temp=0,count=0;
	clrscr();
	label1:
	printf("Enter how many numbers you want to sort");
	scanf("%d",&count); //10-user wants to enter 10 numbers which will be placed at 0-9 index at array a;
	if(count>size)
	{
		printf("\n sorry! I can accept only 30 numbers! Please try again");
		goto label1;
	}

	for(i=0;i<=count-1;i++)  //This loop makes user give input in array a.
	{
		scanf("%d",&a[i]);
	}   

	for(i=0;i<=count-1;i++) //1    //This double loop performs sorting task
	{
		for(j=i+1;j<=count-1;j++) //2-9
		{
			if(a[i]<a[j])  (Descending always <, ascending >)
			{
				temp=a[i];
				a[i]=a[j];
				a[j]=temp;	
			}
		}
	}
	printf("The highest value is %d",a[0]);
	printf("\n The lowest value is %d",a[count-1]);

getch();
}

void main()
{
	int a[10]={24,54,62,1,35,343,2,647,3,61};
	int high=0,low=0,i=0;
	clrscr();
	low=a[0];   //High is assigned the first value of array a which is 24.
	high=a[0];
	for(i=1;i<=9;i++)
	{
		if(low>a[i])
		{
			low=a[i];
		}
		if(high<a[i])
		{
			high=a[i];
		}
	}
	printf("The lowest value in the series is %d",low);
	printf("The highest value in the series is %d",high);
getch();
}
