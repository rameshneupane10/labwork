
1. Program to find simple interest and compound interest by using function.
	
a. with return, with parameter.

		
#include<stdio.h>   // 
#include<conio.h>
#include<math.h>

float SI(float,float,float);
float CI(float,float,float);

void main()
{
	float p,r,t,cresult,sresult;
	clrscr();
	printf("Enter principal, rate and time");
	scanf("%f %f %f",&p,&r,&t);
	sresult=SI(p,r,t);
	cresult=CI(p,r,t);
	printf("\n Simple interest= %f",sresult);
	printf("\n Compound interest= %f,cresult);
	getch();
}

float SI(float x,float y,float z)
{
     float result;
     result=(x*y*z)/100;
     return result;
}

float CI(float m,float n,float o)
{
    float g;
	
    g=m*(pow((1+n/100),o)-1);
    return g;	

}

Type Conversion in C programming:

Type casting refers to changing an variable of one data type into another.

a. Implicit Type Conversion
		When the type conversion is performed automatically by the compiler without programmers intervention, such type of conversion is known as implicit type conversion
		
eg:

int x;
for(x=97; x<=122; x++)
{
    printf("%c", x);   /*Implicit casting from int to char thanks to %c*/
}  


output: a b c d e f g h and so on till z.// 97 98 till 122



b. Explicit Type Conversion:

The type conversion performed by the programmer by posing the data type of the expression of specific type is known as explicit type conversion.
The explicit type conversion is also known as "type casting".


int x;
for(x=97; x<=122; x++)
{
    printf("%c", (char)x);   /*Explicit casting from int to char*/
}

#include<conio.h>

{
	int no=0,rem=0,rev=0,temp=0;
	clrscr();
	printf("Enter a number");
	scanf("%d",&no);
	temp=no;
	while(no!=0)
  	{
		rem=no%10;
		rev=rev*10+rem;
		no=no/10;
        }
	if(rev==temp)
	{
		printf("\nThe given number is a palindrome number");
	}
	else
	{
		printf("\n The given number is not a palindrome number");
	}
	getch();
}
