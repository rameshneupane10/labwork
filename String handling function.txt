5.toupper(): #include<ctype.h>

The toupper() function converts a lowercase alphabet to an uppercase alphabet.


program:
#include <stdio.h>
#include <ctype.h>
int main() {
    char c;

    c = 'm';
    printf("%c -> %c", c, toupper(c));

    c = 'D';
    printf("\n%c -> %c", c, toupper(c));

    c = '9';
    printf("\n%c -> %c", c, toupper(c));
    return 0;
}

Output:
m -> M
D -> D
9 -> 9
     

6.tolower(): #include<ctype.h>

The tolower() function converts a uppercase alphabet to an lowercase alphabet.


program:
#include <stdio.h>
#include <ctype.h>
int main() {
    char c;

    c = 'm';
    printf("%c -> %c", c, tolower(c));

    c = 'D';
    printf("\n%c -> %c", c, tolower(c));

    c = '9';
    printf("\n%c -> %c", c, tolower(c));
    return 0;
}

Output:
m -> m
D -> d
9 -> 9

9. strcat():
   This function is used to concatenate two strings (combine two strings).
